using System;
using System.Collections.Generic;
using System.Text;

namespace Parlamento
{
    class Program
    {
        static int cant, n, suma, i, veces;
        static List<int> grupos;
        static void Metodo(bool b)
        {
            suma = 0;
            int aux = 0;
            i = 2;
            grupos = new List<int>();
            while (true)
            {
                grupos.Add(i);
                suma += i++;
                aux++;
                if (b)
                {
                    veces++;
                    if (suma >= n)
                        break;
                }
                else if (aux == veces)
                    break;
            }
            suma -= n;
        }
        static void Main(string[] args)
        {
            bool cond;
            string sol;
            cant = int.Parse(Console.ReadLine());
            while (cant-- > 0)
            {
                n = int.Parse(Console.ReadLine());
                veces = 0;
                Metodo(true);
                cond = true;
                if (suma != 0)
                {
                    while (cond)
                    {
                        for (int j = 0; j < grupos.Count; j++)
                            if (grupos[j] == suma)
                            {
                                grupos.RemoveAt(j);
                                cond = false;
                            }
                        if (grupos.Count == 0)
                        {
                            veces++;
                            Metodo(false);
                        }
                        else if (cond)
                        {
                            suma -= grupos[0];
                            grupos.RemoveAt(0);
                        }
                    }
                }
                sol = "";
                for (int j = 0; j < grupos.Count; j++)
                    sol += grupos[j] + " ";
                Console.WriteLine(sol.Trim());
            }
        }
    }
}
