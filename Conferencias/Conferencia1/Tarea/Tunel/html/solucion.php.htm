using System;
using System.Collections.Generic;
using System.Text;

namespace Tunel
{
    class Program
    {
        static void Main(string[] args)
        {
            int cases = int.Parse(Console.ReadLine()), cars,i;
            string[] ids;
            List<string> speedUp = new List<string>();
            string id;
            while (cases-- > 0)
            {
                cars = int.Parse(Console.ReadLine());
                ids = new string[cars];
                speedUp.Clear();
                for (i = 0; i < cars; i++)
                    ids[i] = Console.ReadLine();
                i = 0;
                for (;cars > 0; cars--)
                {
                    id = Console.ReadLine();
                    if (ids[i] != id)
                    {
                        if (!speedUp.Contains(id))
                            speedUp.Add(id);
                    }
                    else
                        while (++i < ids.Length && speedUp.Contains(ids[i])) ;
                }
                Console.WriteLine(speedUp.Count);
            }
        }
    }
}
