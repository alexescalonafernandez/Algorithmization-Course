using System;
using System.Collections.Generic;
using System.Text;

namespace Lexical
{
    class Program
    {
        static int Factorial(int n)
        {
            int f = 1;
            while (n > 1)
                f *= n--;
            return f;
        }

        static void Main(string[] args)
        {
            int[] factorial = new int[13];
            factorial[0] = factorial[1] = 1;
            for (int i = 2; i < 13; i++)
                factorial[i] = i * factorial[i - 1];
            int cant = int.Parse(Console.ReadLine());
            int[] num;
            string[] arr;
            while (cant-- > 0)
            {
                int n = int.Parse(Console.ReadLine());
                arr = Console.ReadLine().Split(' ');
                num = new int[n];
                int suma = 0;
                for (int i = 0; i < n; i++)
                    num[i] = int.Parse(arr[i]);
                for (int i = 0; i <n; i++)
                {
                    int f = factorial[n - i - 1];
                    int veces = 0;
                    for (int j = i + 1; j < n; j++)
                        if (num[i] > num[j])
                            veces++;
                    suma += veces * f;

                }
                Console.WriteLine(suma + 1);
            }
        }
    }
}
